### 支付回调核心代码（唯一要严谨的部分）

```java
@PostMapping("/notify/wechat")
public String handleWechatNotify(@RequestBody String body, HttpServletRequest request) {
    // 1. 验签 + 解密（SDK 自动完成）
    Transaction transaction = notificationParser.parse(request, body, Transaction.class);
    
    // 2. 幂等检查
    String orderNo = transaction.getOutTradeNo();
    Order order = orderService.getByOrderNo(orderNo);
    if (order == null || order.getStatus() == OrderStatus.PAID) {
        return "SUCCESS"; // 已处理或不存在，直接返回
    }
    
    // 3. 事务处理
    orderService.handlePaySuccess(orderNo, transaction.getTransactionId());
    // 内部：更新订单状态 + 加积分 + 写流水（同一事务）
    
    return "SUCCESS"; // ⚠️ 必须返回，否则微信重试24小时
}
```

### 退款积分扣回逻辑

```java
@Transactional
public void handleRefund(String orderNo, int refundAmount) {
    Order order = orderService.getByOrderNo(orderNo);
    int pointsToDeduct = order.getPoints(); // 该订单对应的积分数
    
    UserPoints userPoints = pointsService.getByUserId(order.getUserId());
    
    if (userPoints.getBalance() >= pointsToDeduct) {
        // 余额够扣 → 直接扣
        pointsService.changePoints(order.getUserId(), -pointsToDeduct, 
            "REFUND_DEDUCT", orderNo, "订单退款扣回积分");
    } else {
        // 余额不够 → 扣到 0（方案A：差额不追究）
        pointsService.changePoints(order.getUserId(), -userPoints.getBalance(),
            "REFUND_DEDUCT", orderNo, "订单退款扣回积分（余额不足，扣至0）");
    }
    
    // 调用微信退款 API
    refundService.refund(orderNo, refundAmount);
    // 更新订单状态
    orderService.updateStatus(orderNo, OrderStatus.REFUNDED);
}
```

### 数据库表设计

```sql
-- 积分账户表（带乐观锁）
CREATE TABLE `user_points` (
  `id`            bigint PRIMARY KEY AUTO_INCREMENT,
  `user_id`       bigint UNIQUE NOT NULL COMMENT '用户ID',
  `balance`       int DEFAULT 0 COMMENT '当前积分余额',
  `total_earned`  int DEFAULT 0 COMMENT '累计获得积分',
  `total_spent`   int DEFAULT 0 COMMENT '累计消耗积分',
  `version`       int DEFAULT 1 COMMENT '乐观锁版本号',
  `create_time`   datetime DEFAULT CURRENT_TIMESTAMP,
  `update_time`   datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) COMMENT '用户积分账户';

-- 积分流水表（不可修改，永久保留）
CREATE TABLE `points_flow` (
  `id`            bigint PRIMARY KEY AUTO_INCREMENT,
  `user_id`       bigint NOT NULL COMMENT '用户ID',
  `flow_type`     tinyint NOT NULL COMMENT '1=收入 2=支出',
  `biz_type`      varchar(32) NOT NULL COMMENT 'RECHARGE/GENERATE/REGISTER/INVITE/REFUND/ADMIN',
  `biz_id`        varchar(64) COMMENT '关联业务ID（订单号等）',
  `amount`        int NOT NULL COMMENT '变动积分数（正数）',
  `balance_after` int NOT NULL COMMENT '变动后余额',
  `operator`      varchar(64) COMMENT '操作人（管理员操作时）',
  `remark`        varchar(255) COMMENT '备注',
  `create_time`   datetime DEFAULT CURRENT_TIMESTAMP,
  INDEX idx_user_time (`user_id`, `create_time`),
  INDEX idx_biz (`biz_type`, `biz_id`)
) COMMENT '积分流水记录';
```

### 核心函数（所有场景共用）

```java
/**
 * 积分变动统一入口
 * amount > 0 增加，< 0 扣减
 * 黄金原则：每次变动必须同时写流水，在同一个事务中
 */
@Transactional(rollbackFor = Exception.class)
public void changePoints(Long userId, int amount, String bizType, 
                          String bizId, String remark, String operator) {
    // 1. 乐观锁更新余额
    UserPoints points = pointsMapper.selectByUserId(userId);
    int newBalance = points.getBalance() + amount;
    if (newBalance < 0) {
        throw new BusinessException("积分余额不足");
    }
    
    int rows = pointsMapper.updateBalance(userId, amount, newBalance, points.getVersion());
    if (rows == 0) {
        throw new OptimisticLockException("并发冲突，请重试");
    }
    
    // 2. 写流水（不可删除、不可修改）
    PointsFlow flow = new PointsFlow();
    flow.setUserId(userId);
    flow.setFlowType(amount > 0 ? 1 : 2);
    flow.setBizType(bizType);
    flow.setBizId(bizId);
    flow.setAmount(Math.abs(amount));
    flow.setBalanceAfter(newBalance);
    flow.setOperator(operator);
    flow.setRemark(remark);
    flowMapper.insert(flow);
}

// 使用示例：
// 充值：  changePoints(userId, +100, "RECHARGE", orderId, "购买100积分套餐", null)
// 生成：  changePoints(userId, -10,  "GENERATE", taskId, "生成PPT: xxx", null)
// 注册：  changePoints(userId, +20,  "REGISTER", null, "新用户注册赠送", null)
// 邀请：  changePoints(userId, +10,  "INVITE", inviteId, "邀请奖励", null)
// 退款：  changePoints(userId, -100, "REFUND", orderId, "订单退款扣回", null)
// 管理员：changePoints(userId, +50,  "ADMIN", null, "管理员手动补发", "admin_张三")
```
